#make a dictionary of all of the countries by their regional gdp page

urls = {
'Africa' : "https://en.wikipedia.org/wiki/List_of_African_countries_by_GDP_(nominal)",
'Asia-Pacific' : "https://en.wikipedia.org/wiki/List_of_countries_in_Asia-Pacific_by_GDP_(nominal)",
'Europe' : "https://en.wikipedia.org/wiki/List_of_sovereign_states_in_Europe_by_GDP_(nominal)",
'Arab League' : "https://en.wikipedia.org/wiki/List_of_Arab_League_countries_by_GDP_(nominal)",

'North America' : "https://en.wikipedia.org/wiki/List_of_North_American_countries_by_GDP_(nominal)",
'Oceania' : "https://en.wikipedia.org/wiki/List_of_Oceanian_countries_by_GDP_(nominal)",
}

#loop through url dictionary
for url in urls:

  ua = {

    print(f"Processing region: {region} from {url}")
    # Use requests to get the response, passing the headers
    page = rq.get(url, headers=ua)
    page.raise_for_status() # Raise an exception for bad status codes

    bs4page = bs4.BeautifulSoup(page.text, 'html.parser')

    tables = bs4page.find_all('table', {'class': "wikitable"})

    gdp = pd.read_html(StringIO(str(tables[0])))[0]
    gdp = gdp.dropna()

    gdp['Region'] = region
    #add each df to the master df
    all_regions_gdp.append(gdp)

# Concatenate all regional dataframes into a single dataframe
combined_gdp_df = pd.concat(all_regions_gdp, ignore_index=True)

#Now we need to link the levels from the smallest to the next biggest, starting from total ICV 
#we'll create a function to do this so it can be applied easily to each level
def agg_links(df, parent_col, child_col, val_col):
    tmp = df[[parent_col, child_col, val_col]].dropna(subset=[parent_col, child_col])
    tmp = tmp.groupby([parent_col, child_col], as_index=False)[val_col].sum()
    return tmp.rename(columns={parent_col: "source", child_col: "target", val_col: "value"})

#create a list defining how the function will be used to connect each of the levels
links_list = [
    agg_links(subjectData.assign(ICV="ICV"), "ICV", "level1", value_col),
    agg_links(subjectData, "level1", "level2", value_col),
    agg_links(subjectData, "level2", "level3", value_col),
    agg_links(subjectData, "level3", "level4", value_col),
    agg_links(subjectData, "level4", "roi", value_col)
]

#create a df of the connections for ease of plotting
links_df = pd.concat(links_list, ignore_index=True)
links_df = links_df[links_df["value"] > 0].copy()

#define a list of all distinct regions at each level to map from
all_nodes = pd.unique(pd.concat([links_df["source"], links_df["target"]], ignore_index=True))
label_to_idx = {label: i for i, label in enumerate(all_nodes)}
links_df["source_idx"] = links_df["source"].map(label_to_idx)
links_df["target_idx"] = links_df["target"].map(label_to_idx)

# Define node levels (ICV, 0 to ROI, 5) for Sankey flow organization
node_level = {}
def set_level(labels, level):
    for label in labels:
        node_level[label] = min(node_level.get(label, level), level) if label in node_level else level

#use the new links_df to set the ICV as the source of the Sankey, and progressively define each level
set_level(links_df[links_df["source"] == "ICV"]["target"].unique(), 1)
set_level(subjectData["level2"].unique(), 2)
set_level(subjectData["level3"].unique(), 3)
set_level(subjectData["level4"].unique(), 4)
set_level(subjectData["roi"].unique(), 5)
node_level["ICV"] = 0

n: int = ..., Return the first `n` rows.

This function returns the first `n` rows for the object based
on position. It is useful for quickly testing if your object
has the right type of data in it.

For negative values of `n`, this function returns all rows except  
the last `|n|` rows, equivalent to `df[:n]`.

If n is larger than the number of rows, this function returns all rows.

Parameters
----------
n : int, default 5  
&nbsp;&nbsp;&nbsp;&nbsp;Number of rows to select.

Returns
-------
same type as caller  
&nbsp;&nbsp;&nbsp;&nbsp;The first `n` rows of the caller object.

See Also
--------
DataFrame.tail: Returns the last `n` rows.
